<script>

function hearts(){
  for(let i=0;i<60;i++){
    let h=document.createElement("div");
    h.innerHTML="💖";
    h.className="heart";
    h.style.left=Math.random()*100+"vw";
    h.style.animationDuration=(2+Math.random()*4)+"s";
    document.body.appendChild(h);
  }
}

let message =
"💌 رُبى...\n\n" +
"أنا ما بحبك حب عادي...\n" +
"أنا بموت فيكي وبعشقك 💖\n\n" +
"إنتِ مش شخص...\n" +
"إنتِ احلى شي بحياتي 💫\n\n" +
"كل يوم بحبك أكتر...\n\n" +

"وبكل صدق...\n" +
" أنا بحبك ورح ضل حبك لآخر يوم بعمري وما رح يكون في غيرك بقلبي❤️";

function start(){
  document.getElementById("codeBox").style.display="block";
}

function type(){
  hearts();
  let i=0;
  let box=document.getElementById("msg");
  box.innerHTML="";

  function typing(){
    if(i<message.length){
      box.innerHTML += message.charAt(i);
      i++;
      setTimeout(typing,35);
    }
  }

  typing();
}

function check(){
  let code=document.getElementById("code").value.trim();
  let error=document.getElementById("error");

  if(code !== "1542004"){
    error.innerText="❌ كلمة السر غلط 💔";
    return;
  }

  error.innerText="";

  // 🎬 صوت سينمائي
  let audio = document.getElementById("cinemaSound");
  audio.currentTime = 0;
  audio.play();

  type();
}

// 💓 نبض القلب مع صوت

  sound.currentTime=0;
  sound.play();
